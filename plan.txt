  "manifest_version": 3,
  "name": "Google Search Cleaner",
  "version": "1.0",
  "permissions": ["storage", "scripting"],

2. The Content Script (content.js)

This script runs directly on the Google search page. It looks for the AI box (usually wrapped in a specific div) and hides it.
JavaScript

// Function to hide AI and add the toggle button
function cleanSearch() {
  chrome.storage.sync.get(['hideAI', 'hideForums'], (prefs) => {
    // 1. Identify the AI Overview container (ID/Class changes, so we target likely containers)
    const aiBox = document.querySelector('div[data-asynctype="du_kg"], .kp-blk, [data-attrid="wa:/description"]');

    if (aiBox && prefs.hideAI) {
      aiBox.style.display = 'none';

      // Create "See what AI said" button
      if (!document.getElementById('show-ai-btn')) {
        const btn = document.createElement('button');
        btn.id = 'show-ai-btn';
        btn.innerText = 'See what AI said';
        btn.style = 'margin: 10px; padding: 5px 10px; cursor: pointer;';
        btn.onclick = () => { aiBox.style.display = 'block'; btn.remove(); };
        aiBox.parentNode.insertBefore(btn, aiBox);
      }
    }

    // 2. Filter Forum Posts (Discussions)
    if (prefs.hideForums) {
      const forumBlocks = document.querySelectorAll('div:has(h3:contains("Discussions")), [data-attrid="forum"]');
      forumBlocks.forEach(el => el.style.display = 'none');
    }
  });
}

// Run on load and whenever the page updates (AJAX)
const observer = new MutationObserver(cleanSearch);
observer.observe(document.body, { childList: true, subtree: true });
cleanSearch();
